# 执行历史

执行历史是智能终端系统的记录组件，负责记录用户的所有命令执行历史，包括用户输入、推荐的命令、实际执行的命令、执行结果等信息。历史记录功能支持用户查询、统计和重复执行之前的命令。

## 什么是执行历史？

执行历史是一个持久化的日志系统，记录了系统中发生的所有命令执行事件。它提供了完整的审计轨迹，帮助用户回顾操作历史，并且可以作为命令推荐系统的基础数据，用于学习和改进推荐算法。

**关键特征**:
- 完整记录：记录命令执行的完整流程
- 持久化存储：数据保存到文件，系统重启后仍然可用
- 查询支持：支持多种查询方式（时间、命令类型、关键词等）
- 自动清理：自动清理过期的历史记录，控制存储大小

## 结构

### 历史记录模型

```python
from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional
from enum import Enum

class RecordStatus(Enum):
    SUCCESS = "success"          # 执行成功
    FAILED = "failed"            # 执行失败
    TIMEOUT = "timeout"          # 执行超时
    CANCELLED = "cancelled"      # 用户取消

@dataclass
class RecommendedCommand:
    """推荐的命令"""
    command_id: str              # 命令 ID
    command: str                 # 完整命令行
    similarity: float            # 相似度分数
    selected: bool              # 是否被用户选择

@dataclass
class HistoryRecord:
    """历史记录"""
    id: str                      # 记录唯一标识符
    user_input: str              # 用户原始输入
    recommended_commands: List[RecommendedCommand]  # 推荐的命令
    executed_command: Optional[str] = None          # 实际执行的命令
    exit_code: Optional[int] = None                # 退出码
    output: Optional[str] = None                   # 命令输出
    error: Optional[str] = None                    # 错误信息
    status: RecordStatus = RecordStatus.SUCCESS     # 执行状态
    model_used: str = ""                           # 使用的 AI 模型
    duration_ms: int = 0                           # 执行时长（毫秒）
    timestamp: datetime = None                      # 执行时间戳
```

### 历史记录存储

```python
from typing import List
from pathlib import Path

class HistoryStorage:
    """历史记录存储"""

    def __init__(self, file_path: str = "data/history.json"):
        self.file_path = Path(file_path)
        self._ensure_directory()

    async def load(self) -> List[HistoryRecord]:
        """加载所有历史记录"""
        if not self.file_path.exists():
            return []

        async with aiofiles.open(self.file_path, 'r') as f:
            content = await f.read()
            data = json.loads(content)
            return [HistoryRecord(**record) for record in data.get('records', [])]

    async def save(self, records: List[HistoryRecord]):
        """保存历史记录"""
        data = {
            'records': [record.__dict__ for record in records],
            'version': '1.0',
            'updated_at': datetime.now().isoformat()
        }

        async with aiofiles.open(self.file_path, 'w') as f:
            await f.write(json.dumps(data, indent=2))

    async def append(self, record: HistoryRecord):
        """追加单条记录"""
        records = await self.load()
        records.append(record)
        await self.save(records)

    def _ensure_directory(self):
        """确保目录存在"""
        self.file_path.parent.mkdir(parents=True, exist_ok=True)
```

### 历史记录管理器

```python
from typing import List, Optional
from datetime import datetime, timedelta

class HistoryRecorder:
    """历史记录管理器"""

    def __init__(self, storage: HistoryStorage, max_records: int = 1000):
        self.storage = storage
        self.max_records = max_records

    async def record(self, history_data: dict) -> HistoryRecord:
        """记录一次命令执行"""
        # 创建记录对象
        record = HistoryRecord(
            id=self._generate_id(),
            user_input=history_data.get('user_input', ''),
            recommended_commands=history_data.get('recommended_commands', []),
            executed_command=history_data.get('executed_command'),
            exit_code=history_data.get('exit_code'),
            output=history_data.get('output'),
            error=history_data.get('error'),
            status=RecordStatus(history_data.get('status', 'success')),
            model_used=history_data.get('model_used', ''),
            duration_ms=history_data.get('duration_ms', 0),
            timestamp=datetime.now()
        )

        # 保存记录
        await self.storage.append(record)

        # 清理过期记录
        await self._cleanup_old_records()

        return record

    async def query(self, **filters) -> List[HistoryRecord]:
        """查询历史记录"""
        records = await self.storage.load()
        return HistoryQuery(records).filter(**filters)

    async def get_by_id(self, record_id: str) -> Optional[HistoryRecord]:
        """根据 ID 获取记录"""
        records = await self.storage.load()
        for record in records:
            if record.id == record_id:
                return record
        return None

    async def get_recent(self, limit: int = 10) -> List[HistoryRecord]:
        """获取最近的记录"""
        records = await self.storage.load()
        return sorted(
            records,
            key=lambda r: r.timestamp,
            reverse=True
        )[:limit]

    async def _cleanup_old_records(self):
        """清理过期记录"""
        records = await self.storage.load()

        # 如果超过最大数量，删除最旧的记录
        if len(records) > self.max_records:
            records = sorted(records, key=lambda r: r.timestamp)
            records_to_keep = records[-self.max_records:]
            await self.storage.save(records_to_keep)

    def _generate_id(self) -> str:
        """生成唯一 ID"""
        return f"hist-{int(datetime.now().timestamp() * 1000)}-{uuid.uuid4().hex[:8]}"
```

### 历史记录查询

```python
from typing import List
from datetime import datetime

class HistoryQuery:
    """历史记录查询"""

    def __init__(self, records: List[HistoryRecord]):
        self.records = records

    def filter(self, **filters) -> List[HistoryRecord]:
        """应用过滤器"""
        result = self.records

        # 按时间范围过滤
        if 'start_time' in filters:
            result = [r for r in result if r.timestamp >= filters['start_time']]
        if 'end_time' in filters:
            result = [r for r in result if r.timestamp <= filters['end_time']]

        # 按命令类型过滤
        if 'command_type' in filters:
            result = [r for r in result if self._get_command_type(r) == filters['command_type']]

        # 按状态过滤
        if 'status' in filters:
            result = [r for r in result if r.status == filters['status']]

        # 按关键词过滤
        if 'keyword' in filters:
            keyword = filters['keyword'].lower()
            result = [r for r in result if keyword in r.user_input.lower()]

        # 按模型过滤
        if 'model_used' in filters:
            result = [r for r in result if r.model_used == filters['model_used']]

        return result

    def _get_command_type(self, record: HistoryRecord) -> str:
        """获取命令类型"""
        if record.executed_command:
            # 这里可以根据实际逻辑判断命令类型
            return "system"  # 或 "custom"
        return "unknown"
```

### 关键字段

| 字段 | 类型 | 描述 | 约束 |
|------|------|------|------|
| `id` | `str` | 记录唯一标识符 | 必须唯一 |
| `user_input` | `str` | 用户原始输入 | 不为空 |
| `status` | `RecordStatus` | 执行状态 | success/failed/timeout/cancelled |
| `timestamp` | `datetime` | 执行时间戳 | 不为空 |

## 不变量

这些规则对有效的历史记录必须始终成立：

1. **ID 唯一性**: 每条历史记录的 ID 必须唯一
   - 使用时间戳 + UUID 生成

2. **时间顺序**: 记录的时间戳必须反映实际的执行时间
   - 不能手动修改时间戳

3. **最大数量限制**: 历史记录总数不超过 `max_records`
   - 超过时自动删除最旧的记录

4. **数据完整性**: 记录必须包含所有必要字段
   - user_input, status, timestamp 必须有值

```python
async def get_statistics(recorder: HistoryRecorder) -> dict:
    """获取历史记录统计信息"""
    records = await recorder.storage.load()

    total = len(records)
    success = len([r for r in records if r.status == RecordStatus.SUCCESS])
    failed = len([r for r in records if r.status == RecordStatus.FAILED])

    # 常用命令
    from collections import Counter
    commands = [r.executed_command for r in records if r.executed_command]
    top_commands = Counter(commands).most_common(10)

    # 使用的模型
    models = [r.model_used for r in records if r.model_used]
    model_usage = Counter(models)

    return {
        'total': total,
        'success': success,
        'failed': failed,
        'success_rate': success / total if total > 0 else 0,
        'top_commands': top_commands,
        'model_usage': dict(model_usage)
    }

stats = await get_statistics(recorder)
print(f"总记录数: {stats['total']}")
```

```python
import aiofiles

class HistoryStorage:
    async def load_batch(self, offset: int = 0, limit: int = 100) -> List[HistoryRecord]:
        """批量加载记录"""
        if not self.file_path.exists():
            return []

        async with aiofiles.open(self.file_path, 'r') as f:
```

### 索引优化

```python
from collections import defaultdict

class HistoryRecorder:
    def __init__(self, storage: HistoryStorage, max_records: int = 1000):
        self.storage = storage
        self.max_records = max_records
        self._index = defaultdict(list)
        self._index_built = False

    async def _ensure_index(self):
        """确保索引已构建"""
        if self._index_built:
            return

        records = await self.storage.load()
        for record in records:
            # 按命令索引
            if record.executed_command:
                self._index['command'][record.executed_command].append(record.id)

            # 按状态索引
            self._index['status'][record.status].append(record.id)

        self._index_built = True

    async def query(self, **filters) -> List[HistoryRecord]:
        """使用索引查询"""
        await self._ensure_index()

        if 'command' in filters and 'command' in self._index:
            # 使用索引快速查找
            record_ids = self._index['command'].get(filters['command'], [])
            return [await self.get_by_id(id) for id in record_ids]

        # 否则使用常规查询
        records = await self.storage.load()
        return HistoryQuery(records).filter(**filters)
```

### 添加数据分析功能

```python
class HistoryAnalyzer:
    """历史记录分析器"""

    def __init__(self, recorder: HistoryRecorder):
        self.recorder = recorder

    async def analyze_command_frequency(self) -> dict:
        """分析命令使用频率"""
        records = await self.recorder.storage.load()
        commands = [r.executed_command for r in records if r.executed_command]

        from collections import Counter
        frequency = Counter(commands)

        return {
            'most_common': frequency.most_common(10),
            'total_unique': len(frequency),
            'total_executions': len(commands)
        }

    async def analyze_model_performance(self) -> dict:
        """分析模型性能"""
        records = await self.recorder.storage.load()

        model_stats = defaultdict(lambda: {
            'total': 0,
            'success': 0,
            'avg_duration_ms': 0
        })

        for record in records:
            model = record.model_used
            stats = model_stats[model]
            stats['total'] += 1
            if record.status == RecordStatus.SUCCESS:
                stats['success'] += 1
            stats['avg_duration_ms'] = (
                stats['avg_duration_ms'] * (stats['total'] - 1) + record.duration_ms
            ) / stats['total']

        return dict(model_stats)

    async def analyze_user_patterns(self) -> dict:
        """分析用户使用模式"""
        records = await self.recorder.storage.load()

        # 按小时统计
        hourly_usage = [0] * 24
        for record in records:
            hour = record.timestamp.hour
            hourly_usage[hour] += 1

        # 按星期统计
        weekday_usage = [0] * 7
        for record in records:
            weekday = record.timestamp.weekday()
            weekday_usage[weekday] += 1

        return {
            'hourly_usage': hourly_usage,
            'weekday_usage': weekday_usage,
            'peak_hour': hourly_usage.index(max(hourly_usage)),
            'peak_weekday': weekday_usage.index(max(weekday_usage))
        }
```

### 添加导出功能

```python
import csv
from pathlib import Path

class HistoryExporter:
    """历史记录导出器"""

    def __init__(self, recorder: HistoryRecorder):
        self.recorder = recorder

    async def export_to_csv(self, output_path: str, **filters):
        """导出到 CSV 文件"""
        records = await self.recorder.query(**filters)

        output_path = Path(output_path)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        with open(output_path, 'w', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)

            # 写入表头
            writer.writerow([
                'ID', 'Timestamp', 'User Input', 'Executed Command',
                'Exit Code', 'Status', 'Duration (ms)', 'Model Used'
            ])

            # 写入数据
            for record in records:
                writer.writerow([
                    record.id,
                    record.timestamp.isoformat(),
                    record.user_input,
                    record.executed_command,
                    record.exit_code,
                    record.status.value,
                    record.duration_ms,
                    record.model_used
                ])

    async def export_to_json(self, output_path: str, **filters):
        """导出到 JSON 文件"""
        records = await self.recorder.query(**filters)

        output_path = Path(output_path)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        data = {
            'records': [r.__dict__ for r in records],
            'exported_at': datetime.now().isoformat()
        }

        async with aiofiles.open(output_path, 'w') as f:
            await f.write(json.dumps(data, indent=2))
```
